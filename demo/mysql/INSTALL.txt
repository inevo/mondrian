
Importing the Foodmart data set in MySQL
----------------------------------------

C:\mysql\bin> mysql

mysql> GRANT ALL PRIVILEGES ON *.* TO foodmart@localhost IDENTIFIED BY 'foodmart';
Query OK, 0 rows affected (0.06 sec)

mysql> create database foodmart;
Query OK, 1 row affected (0.00 sec)

mysql> \q

C:\mysql\bin> mysql foodmart < FoodMartTables.sql

C:\mysql\bin> mysql foodmart < FoodMartIndexes.sql

C:\mysql\bin> mysql foodmart < FoodMartData.sql
